CREATE TABLE sites (
    id SERIAL PRIMARY KEY,
    site_code TEXT UNIQUE NOT NULL,
    site_name VARCHAR(50),
    region VARCHAR(50),
    province VARCHAR(50),
    amphur VARCHAR(50),
    tambol VARCHAR(50),
    latitude  NUMERIC(10,7),
    longitude NUMERIC(10,7)
);

CREATE TABLE systems (
    system_id   SERIAL PRIMARY KEY,
    system_name TEXT UNIQUE NOT NULL -- เช่น 2G, 3G, 4G, 5G
);

CREATE TABLE operators (
    operator_id   SERIAL PRIMARY KEY,
    operator_name TEXT UNIQUE NOT NULL -- AIS, TRUE, NT
);

CREATE TABLE site_config (
    id           BIGSERIAL PRIMARY KEY,
    site_id      BIGINT NOT NULL REFERENCES sites(id) ON DELETE CASCADE,
    system_id    INT NOT NULL REFERENCES systems(system_id),
    operator_id  INT NOT NULL REFERENCES operators(operator_id),

    created_at   TIMESTAMPTZ DEFAULT NOW(),

    CONSTRAINT uq_site_system_operator
    UNIQUE (site_id, system_id, operator_id)
);
CREATE INDEX idx_site_config_site
ON site_config (site_id);

CREATE INDEX idx_site_config_full
ON site_config (site_id, system_id, operator_id);

CREATE TABLE site_availability_daily (
    id              BIGSERIAL PRIMARY KEY,
    site_config_id  BIGINT NOT NULL,

    snapshot_date   DATE NOT NULL,
    avg_availability NUMERIC(5,2) NOT NULL
        CHECK (avg_availability >= 0 AND avg_availability <= 100),

    created_at      TIMESTAMPTZ DEFAULT NOW(),

    CONSTRAINT uq_config_day UNIQUE (site_config_id, snapshot_date),

    CONSTRAINT fk_site_config
        FOREIGN KEY (site_config_id)
        REFERENCES site_config(id)
        ON DELETE CASCADE
);
CREATE INDEX idx_availability_date
ON site_availability_daily (snapshot_date);

CREATE INDEX idx_availability_config_date
ON site_availability_daily (site_config_id, snapshot_date DESC);

-- =========================
-- INCIDENTS
-- =========================
CREATE TABLE incidents (
    incident_id               BIGSERIAL PRIMARY KEY,

    global_id                 TEXT,
    incident_number           TEXT NOT NULL,

    ticket_type               TEXT,
    urgency                   TEXT,
    severity                  TEXT,
    subject                   TEXT,
    classification            TEXT,
    status                    TEXT,

    creator_id                TEXT,
    creator_name              TEXT,
    department                TEXT,
    activity_owner            TEXT,
    activity_status           TEXT,
    external_ticket           TEXT,

    owner                     TEXT,
    owner_group               TEXT,

    failure_class             TEXT,
    problem                   TEXT,
    cause                     TEXT,
    remedy                    TEXT,
    external_system           TEXT,

    fault_datetime            TIMESTAMPTZ,
    create_datetime           TIMESTAMPTZ,
    expected_finish_datetime  TIMESTAMPTZ,
    actual_start_datetime     TIMESTAMPTZ,
    actual_finish_datetime    TIMESTAMPTZ,
    restoration_datetime      TIMESTAMPTZ,

    raw_payload               JSONB,

    created_at                TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at                TIMESTAMPTZ NOT NULL DEFAULT NOW(),

    -- 🔥 KEY FIX: generated column สำหรับ dedup
    dedup_key TEXT GENERATED ALWAYS AS (
        COALESCE(NULLIF(global_id, ''), incident_number)
    ) STORED,

    CONSTRAINT uq_incident_dedup UNIQUE (dedup_key)
);

-- indexes สำหรับ query
CREATE INDEX idx_incident_number
ON incidents (incident_number);

CREATE INDEX idx_incidents_fault_datetime
ON incidents (fault_datetime);

CREATE INDEX idx_incidents_status
ON incidents (status);

CREATE INDEX idx_incidents_time_range
ON incidents (fault_datetime, restoration_datetime);

-- =========================
-- INCIDENT IMPACT
-- =========================
CREATE TABLE incident_site_impact (
    id             BIGSERIAL PRIMARY KEY,
    incident_id    BIGINT NOT NULL REFERENCES incidents(incident_id) ON DELETE CASCADE,

    site_id        BIGINT REFERENCES sites(id) ON DELETE CASCADE,
    site_config_id BIGINT REFERENCES site_config(id) ON DELETE CASCADE,

    impact_start   TIMESTAMPTZ,
    impact_end     TIMESTAMPTZ,

    created_at     TIMESTAMPTZ DEFAULT NOW(),

    CHECK (site_id IS NOT NULL OR site_config_id IS NOT NULL),
    CHECK (impact_end IS NULL OR impact_start <= impact_end)
);

CREATE INDEX idx_incident_time
ON incident_site_impact (impact_start, impact_end);

CREATE INDEX idx_incident_site
ON incident_site_impact (site_id);

CREATE INDEX idx_incident_config
ON incident_site_impact (site_config_id);

-- 🔥 range index (ไม่ใช้ NOW() เพื่อให้ planner optimize ได้)
CREATE INDEX idx_incident_time_range
ON incident_site_impact
USING GIST (tstzrange(impact_start, impact_end));

-- =========================
-- TRIGGER: auto update updated_at
-- =========================
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_update_incidents_updated_at
BEFORE UPDATE ON incidents
FOR EACH ROW
EXECUTE FUNCTION update_updated_at();


DROP TABLE IF EXISTS incident_site_impact;
DROP TABLE IF EXISTS site_availability_daily;
DROP TABLE IF EXISTS site_config;
DROP TABLE IF EXISTS operators;
DROP TABLE IF EXISTS incidents;
DROP TABLE IF EXISTS systems;
