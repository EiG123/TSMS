CREATE TABLE sites (
    id SERIAL PRIMARY KEY,
    site_id VARCHAR(50) UNIQUE,
    region VARCHAR(50),
    province VARCHAR(50)
);

CREATE TABLE pm (
    id SERIAL PRIMARY KEY,
    site_id INTEGER,
    site_name VARCHAR(50),
    type VARCHAR(50),
    date VARCHAR(100),
    round VARCHAR(50),
    planwork VARCHAR(255),
    service_status VARCHAR(50),
    progress_status VARCHAR(50),
    vendor VARCHAR(50),
    remark TEXT,
    
    checkin_time TIMESTAMP,
    checkout_time TIMESTAMP,
    last_activity_at TIMESTAMP,
    status VARCHAR(20) DEFAULT 'checkout',
    auto_checkout BOOLEAN DEFAULT FALSE,
    user_id INTEGER,

    created_by VARCHAR(50),
    created_at VARCHAR(50),
    updated_by VARCHAR(50),
    updated_at VARCHAR(50),
    deleted_at VARCHAR(50),
    deleted_by VARCHAR(50)
);

CREATE TABLE pm_cabinet (
    id BIGSERIAL PRIMARY KEY,
    cabinet_name VARCHAR(50),
    cabinet_network VARCHAR(50),
    rectifier VARCHAR(50),
    pm_id INTEGER NOT NULL,

    CONSTRAINT fk_pm_cabinet
        FOREIGN KEY (pm_id)
        REFERENCES pm(id)
        ON DELETE CASCADE
);

CREATE TABLE pm_rectifier (

    CONSTRAINT fk_pm_rectifier
        FOREIGN KEY (cabinet_id)
        REFERENCES pm_cabinet(id)
        ON DELETE CASCADE
);

CREATE TABLE pm_title (
    id SERIAL PRIMARY KEY,
    title VARCHAR(50),
    description VARCHAR(255),
    key VARCHAR(50),
    type VARCHAR(50),
    status VARCHAR(50) NOT NULL,
    rank INTEGER NOT NULL
);

CREATE TABLE pm_title_child (
    id SERIAL PRIMARY KEY,
    title_id INTEGER NOT NULL,
    title_child_name VARCHAR(100) NOT NULL,
    description TEXT,
    rank INTEGER NOT NULL,
    status VARCHAR(30) NOT NULL,

    value_status_1 VARCHAR(50),
    value_name_1 TEXT,
    value_input_type_1 VARCHAR(50),

    value_status_2 VARCHAR(50),
    value_name_2 TEXT,
    value_input_type_2 VARCHAR(50),

    value_status_3 VARCHAR(50),
    value_name_3 TEXT,
    value_input_type_3 VARCHAR(50),

    img_status VARCHAR(50),

    CONSTRAINT fk_title_child_title
        FOREIGN KEY (title_id)
        REFERENCES pm_title(id)
        ON DELETE CASCADE
);
CREATE INDEX idx_pm_title_child
ON pm_title_child(title_id);

CREATE TABLE pm_details (
    id BIGSERIAL PRIMARY KEY,

    pm_id INTEGER NOT NULL,
    title_id INTEGER NOT NULL,
    title_child_id INTEGER NOT NULL,
    order_number INTEGER NOT NULL,

    value_1 VARCHAR(255),
    value_2 VARCHAR(255),
    value_3 VARCHAR(255),

    CONSTRAINT unique_pm_detail
    UNIQUE (pm_id, title_id, title_child_id, order_number)

    CONSTRAINT fk_pm_details
        FOREIGN KEY (title_child_id)
        REFERENCES pm_title_child(id)
        ON DELETE CASCADE
);
CREATE INDEX idx_pm_details_pm_id
ON pm_details (pm_id);

CREATE TABLE pm_problem (
    id SERIAL PRIMARY KEY,
    pm_id BIGINT,
    severity SMALLINT --1=MAJOR, 2=MINOR

    --LifeCycle
    problem_status SMALLINT NOT NULL,
    
    --Vendor Verify
    verify_status SMALLINT NOT NULL DEFAULT 0,

    --TRUE Audit
    audit_status SMALLINT NOT NULL DEFAULT 0,

    plan_data DATE,
    fixed_at TIMESTAMP,
    closed_at TIMESTAMP,

    is_overdue BOOLEAN DEFAULT FALSE,

    created_by BIGINT,
    updated_by BIGINT,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),

    CONSTRAINT fk_pm_problem
        FOREIGN KEY (pm_id)
        REFERENCES pm(id)
        ON DEFAULT CASCADE
);

CREATE TABLE problem_logs (
    id BIGSERIAL PRIMARY KEY,
    problem_id BIGINT NOT NULL,

    action VARCHAR(50),

    from_problem_status SMALLINT,
    to_problem_status SMALLINT,

    from_verify_status SMALLINT,
    to_verify_status SMALLINT,

    action_by BIGINT,
    action_role VARCHAR(30),

    remark TEXT,
    action_at TIMESTAMP DEFAULT NOW()

    CONSTRAINT fk_pm_problem_logs
        FOREIGN KEY (problem_id)
        REFERENCES pm_problem(id)
        ON DEFAULT CASCADE
);

CREATE TABLE status_master (
    id SMALLINT PRIMARY KEY,
    module VARCHAR(30) NOT NULL,        -- PM / PROBLEM / AUDIT / BATTERY
    layer VARCHAR(30) NOT NULL,         -- PROCESS / VERIFY / AUDIT
    code VARCHAR(50) NOT NULL,          -- DARFT / SUBMITTED / CLOSED
    name VARCHAR(100) NOT NULL,         -- Display Name
    description TEXT,
    is_active BOOLEAN DEFAULT TRUE,
    sort_order INTEGER DEFAULT 0

    UNIQUE(module, layer, code)
);
